rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Portfolio data: Publicly readable for the website
    // Write access is DISALLOWED publicly (must be handled via authorized methods)
    match /portfolio/{document=**} {
      allow read: if true;
      allow write: if false; 
    }

    // Admin Credentials: NO access allowed from the frontend
    // This is now handled securely via server-side API routes
    match /admin-credential/{document=**} {
      allow read: if false; 
      allow write: if false; 
    }
  }
}
